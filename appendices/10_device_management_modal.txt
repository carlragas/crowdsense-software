/// Full-screen modal for managing ESP32 prototype units.
/// Provides CRUD operations for devices and sensor threshold configuration.
class DeviceManagementModal extends StatefulWidget {
  @override
  State<DeviceManagementModal> createState() => _DeviceManagementModalState();
}

    super.initState();
    _listenToSensorData();   // Real-time sensor data for status display
    _listenToDevices();      // Real-time device registry
  }

    _devicesSubscription = _dbRef.child('prototype_units').onValue.listen((event) {

  }

    _sensorDataSubscription = _dbRef.child('sensor_data').onValue.listen((event) {

      // Re-merges into device list for live status display
    });

  // Adds new device to Firebase with initial configuration
  Future<void> _addDeviceToFirebase(String mac, String name) async {
    await _dbRef.child('prototype_units').child(mac).set({
      "name": name, "priority": _devices.length,
      "config": {
        "smoke_threshold": 300.0, "flame_threshold": 200.0,
        "include_in_headcount": true, "sync_count": false,
      }
    });
    // Pre-initialize sensor_data with all Arduino-written fields
    await _dbRef.child('sensor_data').child(mac).set({
      "people_inside": 0, "total_exits": 0, "temperature": 0.0,
      "gas": 0, "flame": 0, "smoke_threshold": 300.0, "flame_threshold": 200.0,
      "siren_alert_active": false, "siren_clear_active": false,
      "power_status": "Unknown",
    });
  }

  // Validates MAC format (XX:XX:XX:XX:XX:XX) and adds device
  void _handleAddDevice() async {
    final macRegex = RegExp(r'^([0-9A-F]{2}[:]){5}([0-9A-F]{2})$');
    if (!macRegex.hasMatch(mac)) { /* show error */ return; }
    await _addDeviceToFirebase(mac, name);
  }

  // Removes device from both prototype_units and sensor_data
  Future<void> _removeDeviceFromFirebase(String mac) async {
    await _dbRef.child('prototype_units').child(mac).remove();
    await _dbRef.child('sensor_data').child(mac).remove();
  }

  // Edits device: updates name, thresholds, or migrates to new MAC
  void _executeEditDevice(String oldMac, String newMac, String newName, Map sensors) async {
    if (oldMac == newMac) {
      // Same MAC: update in place
      await _dbRef.child('prototype_units').child(oldMac).update({"name": newName, "config": sensors});
      await _dbRef.child('sensor_data').child(oldMac).update({/* thresholds */});
    } else {
      // MAC changed: copy data to new key, delete old key
    }
  }
